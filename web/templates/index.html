📄 Inference Example

import onnxruntime as ort
import numpy as np

sess = ort.InferenceSession(
  "checkpoints/best_model.onnx",
  providers=["CUDAExecutionProvider","CPUExecutionProvider"]
)
candles = np.random.randn(1, 30, 11).astype(np.float32)
logits, attn = sess.run(None, {"candles": candles})
label = ["BUY","HOLD","SELL"][logits[0].argmax()]
print(f"Signal: {label}")
